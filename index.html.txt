<script>
  const produtos = [
    {id: 1, nome: "Hambúrguer", preco: 20, img: "https://via.placeholder.com/200x150"},
    {id: 2, nome: "Pizza", preco: 35, img: "https://via.placeholder.com/200x150"},
    {id: 3, nome: "Refrigerante", preco: 8, img: "https://via.placeholder.com/200x150"}
  ];

  const carrinho = [];

  function mostrarProdutos() {
    const container = document.getElementById('produtos');
    container.innerHTML = '';
    produtos.forEach(p => {
      const div = document.createElement('div');
      div.className = 'produto';
      div.innerHTML = `
        <img src="${p.img}" alt="${p.nome}">
        <h4>${p.nome}</h4>
        <p>R$ ${p.preco}</p>
        <button onclick="adicionarAoCarrinho(${p.id})">Adicionar</button>
      `;
      container.appendChild(div);
    });
  }

  function adicionarAoCarrinho(id) {
    const produto = produtos.find(p => p.id === id);
    carrinho.push(produto);
    atualizarCarrinho();
  }

  function removerDoCarrinho(index) {
    carrinho.splice(index, 1);
    atualizarCarrinho();
  }

  function atualizarCarrinho() {
    const carrinhoItens = document.getElementById('carrinho-itens');
    const totalElem = document.getElementById('total');
    carrinhoItens.innerHTML = '';
    let total = 0;
    carrinho.forEach((item, index) => {
      total += item.preco;
      const div = document.createElement('div');
      div.className = 'carrinho-item';
      div.innerHTML = `
        ${item.nome} - R$ ${item.preco} 
        <button onclick="removerDoCarrinho(${index})">X</button>
      `;
      carrinhoItens.appendChild(div);
    });
    totalElem.innerText = total;
  }

  function finalizarPedido() {
    if (carrinho.length === 0) {
      alert("Carrinho vazio!");
      return;
    }
    alert("Pedido finalizado!\nTotal: R$ " + carrinho.reduce((sum, item) => sum + item.preco, 0));
    carrinho.length = 0;
    atualizarCarrinho();
  }

  mostrarProdutos();
</script>

function finalizarPedido() {
    if (carrinho.length === 0) {
        alert("Carrinho vazio!");
        return;
    }
    let mensagem = "Olá! Gostaria de fazer o pedido:\n";
    carrinho.forEach(item => {
        mensagem += `- ${item.nome} R$ ${item.preco}\n`;
    });
    let total = carrinho.reduce((sum, item) => sum + item.preco, 0);
    mensagem += `Total: R$ ${total}`;
    
    // Transformar em link do WhatsApp
    const telefone = "5511999999999"; // Substitua pelo seu número com DDD
    const url = `https://api.whatsapp.com/send?phone=${telefone}&text=${encodeURIComponent(mensagem)}`;
    window.open(url, "_blank");
}
